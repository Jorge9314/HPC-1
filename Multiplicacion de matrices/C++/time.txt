void mult(float* A, int rowsA, int colsA, float* B, int rowsB, int colsB, float* C){
  for(int i = 0; i < rowsA; i++){
    for(int j = 0; j< colsB; j++){
      int suma = 0;
      for(int k = 0; k < rowsB; k++){
        suma = suma + A[i * colsA + k] * B[ k * colsB + j];
      }
      C[i * colsB + j] = suma;
    }
  }
}
